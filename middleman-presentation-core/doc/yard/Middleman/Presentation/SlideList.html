def initialize(names, slide_builder: Slide, **args, &block)
  @slides = Array(names).map { |n| slide_builder.new(n, **args) }
  @transformers = []

  block.call(self) if block_given?

  @slides = transformers.reduce(@slides) { |a, e| e.transform(a) }
end
